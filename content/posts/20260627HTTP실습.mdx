## 기초 코드

```html
 <!--HTML5 표준 형식으로 작성 되었음을 선언(내 타입은 HTML이야)-->
<!DOCTYPE HTML>
<!--메타데이터(제목, 스타일시트 등)을 담는 컨테이너 영역(제목과 스타일같은 처음 보이는 부분을 담당하는 창고)-->
<!--브라우저에 보이지 않는 메타 정보 영역-->
<head>
<!--표준언어를 UTF-8로 선언 글자 깨짐이 덜함(한글 등 유니코드 문자 깨짐 방지)-->
<meta charset="UTF-8">
<!--모바일 반응형을 위한 영역(뷰포트)설정,화면 너비를 기기 너비에 맞추고 초기 배율을 1로 설정-->
<meta name="viewport" content="width=device-width, initial-scale=1.0">
<!--사이트의 제목-->
<!--브라우저 탭에 표시되는 제목-->
<title>자기소개</title>
<!--외부 CSS(style.css파일을 현재 html파일에 연결)-->
<link rel="stylesheet" href="style.css">
</head>
<!--핵심이 되는 본문의 내용-->
<!--화면에 실제로 보이는 콘텐츠 영역-->
<body>
 <!--페이지 상단의 영역-->
 <header>
   <!--네비게이션 메뉴(페이지 내 이동 링크 모음)-->
   <nav>
    <!--인라인 텍스트 요소(로고나 타이틀의 역할)--> 
    <span>내 소개</span>
    <!--순서없는목록(unordered list)-->
    <ul>
        <!--목록 항목 해당 세션으로 이동-->
        <li><a href="#about">소개</a></li>
        <li><a href="#skills">기술</a></li>
        <li><a href="#contact">연락처</a></li>
    </ul>
   </nav> 
 </header>
 <!--페이지의 핵심 콘텐츠 영역-->
 <main>
  <!--네비게이션 링크로 이동-->  
  <section id="about">
  <!--최상위 제목 h1 가장 큰 글자 크기-->
  <h1>안녕하세요</h1>
  <!--문단(단락) 시작-->
  <p>
  ai시대에 있어 코드를 작업한다고 해도 코드의 흐름을 읽을 수 있어야하고 눈으로 봤을 때 어떤 
  코드인지 알아야한다고 생각합니다. ai를 배척하는게 아닌 ai를 활용한 공부, ai와 협력하여 개발을 하겠습니다
  </p>
  </section>
  <section id = "skills">
    <h2>현재 기초부터 배운다는 생각으로 html,css,js 웹 3대 기술을 익히고 있습니다</h2>
  </section>
  <section id = "contact">
    <h2>010-0000-0000</h2>
  </section>
 </main>
 <!--페이지의 하단 영역(저작권, 연락처 등등)-->
 <footer>
   <p>2026 나의 자기소개</p>
 </footer>

</body>
</html>

```
